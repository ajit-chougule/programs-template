class Pattern3
{
    public static void main(String[] args) 
    {
        int no=5;
        for(int i=0 ; i<5 ; i++)
        {
            for(int j=0 ; j<5 ; j++)
            {
                if(i % 2 == 0)
                {
                    System.out.print((i*no+j)+1+" ");
                }
                else
                {
                    System.out.print(((i+1)*no)-j+" ");
                }
            }
            System.out.println();
        }    
    }
}
