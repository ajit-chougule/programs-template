#define MAX_SIZE 1000
using namespace std;

int main()
{
    int arr[MAX_SIZE];
    int size;
    int i, j, temp;

    cout<< "How many numbers you want to sort: ";
    cin>> size;

    cout<< "Enter "<< size <<" elements:"<< endl;
    for(i=0; i < size; i++)
    {
        cin>> arr[i];
    }

    selectionSort(arr,size);

    cout<< "\nElements of array in ascending order: \n";
    for(i=0; i < size; i++)
    {
        cout<< arr[i] << endl;
    }
    return 0;
}

void selectionSort(int arr[], int n)
{
    int i, j, minIndex, tmp;
    for (i = 0; i < n - 1; i++)
    {
        minIndex = i;
        for (j = i + 1; j < n; j++)
        {
          if (arr[j] < arr[minIndex])
              minIndex = j;
        }

        if (minIndex != i)
        {
            tmp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = tmp;
        }
    }
}
